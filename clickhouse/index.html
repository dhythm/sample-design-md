-- テーブルを作成
CREATE TABLE events (
    timestamp DateTime,
    user_id UInt64,
    event_type String,
    properties Map(String, String)
) ENGINE = MergeTree()
ORDER BY (timestamp, user_id);

-- 高速な分析クエリ
SELECT
    event_type,
    count() AS total,
    uniqExact(user_id) AS unique_users
FROM events
WHERE timestamp >= now() - INTERVAL 1 HOUR
GROUP BY event_type
ORDER BY total DESC;
